    # 并发请求

    /**  
 * 并发请求
 * @param {string[]} urls 待请求的 url 数组
 * @param {number} maxNum 最大并发数 
*/
function concurRequest(urls, maxNum) {
  return new Promise(resolve => {
    // 请求数组为空 不发请求 返回空数组结果
    if (urls.length === 0) {
      resolve([])
      return
    }

    const results = []
    let index = 0 // 下一个请求的下标
    let count = 0 // 当前请求完成数量
    // 发送请求
    async function request() {
      // 索引超出数组下标范围
      if (index === urls.length) return
      const i = index
      const url = urls[index]
      index++
      try {
        const resp = await fetch(url)
        // resp加入到results
        results[i] = resp
      } catch (error) {
        // error加入到results
        results[i] = error
      } finally {
        // 判断请求是否全部完成
        count++
        if (count === urls.length) {
          resolve(results)
        }
        request()
      }
    }
    // 最大并发请求数 超过了请求数组的长度
    const times = Math.min(maxNum, urls.length)
    for (let y = 0; y < times; y++) {
      request()
    }
  })
}

    # 使用

    const urls = []
for (let i = 1; i <= 100; i++) {
  urls.push(`https://jsonplaceholder.typicode.com/todos/${i}`)
}
concurRequest(urls, 20).then(reps => {
  console.log(reps)
})
